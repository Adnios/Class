# P1255数楼梯

```c++
#include <bits/stdc++.h>
using namespace std;
long long n,dp[5005][5005],len=1;
void add(int x){
	for(int i=1;i<=len;i++){
		dp[x][i]+=dp[x-1][i]+dp[x-2][i];
		dp[x][i+1]+=dp[x][i]/10;
		dp[x][i]=dp[x][i]%10;
	}
	if(dp[x][len+1]!=0){
		len++;
	}
}
int main(){
	cin>>n;
	dp[1][1]=1;dp[2][1]=2;
	for(int i=3;i<=n;i++){
		add(i);
	}
	for(int i=len;i>=1;i--){
		cout<<dp[n][i];
	}
}
```

# P1601 A+B Problem（高精）

```c++
#include <bits/stdc++.h>
using namespace std;
const int maxn=520;
int a[maxn],b[maxn],c[maxn];
int main(){
    string A,B;
    cin>>A>>B;
    int len=max(A.length(),B.length());
    int j=1;
    for(int i=A.length()-1;i>=0;i--)
        a[j++]=A[i]-'0';
    j=1;
    for(int i=B.length()-1;i>=0;i--)
        b[j++]=B[i]-'0';
    for(int i=1;i<=len;i++){
        c[i]+=a[i]+b[i];
        c[i+1]+=c[i]/10;
        c[i]=c[i]%10;
    }
    if(c[len+1]!=0)
        len++;
    for(int i=len;i>=1;i--)
        printf("%d",c[i]);
}
```
